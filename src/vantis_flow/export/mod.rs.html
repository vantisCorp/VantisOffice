1//! Export module for Vantis Flow
2
3use serde::{Deserialize, Serialize};
4use std::fs::File;
5use std::io::Write;
6
7use crate::{core::Canvas, diagram::DiagramRenderer, FlowError, FlowResult};
8
9/// Export format
10#[derive(Debug, Clone, Serialize, Deserialize, PartialEq)]
11pub enum ExportFormat {
12    Svg,
13    Png,
14    Pdf,
15    Json,
16    Vml,
17    Vantis,
18}
19
20/// Flow exporter
21pub struct FlowExporter {
22    pub options: ExportOptions,
23}
24
25/// Export options
26#[derive(Debug, Clone)]
27pub struct ExportOptions {
28    pub include_background: bool,
29    pub include_grid: bool,
30    pub scale: f64,
31    pub dpi: u32,
32    pub quality: u8,
33}
34
35impl Default for ExportOptions {
36    fn default() -> Self {
37        Self {
38            include_background: true,
39            include_grid: false,
40            scale: 1.0,
41            dpi: 300,
42            quality: 90,
43        }
44    }
45}
46
47impl FlowExporter {
48    pub fn new() -> Self {
49        Self {
50            options: ExportOptions::default(),
51        }
52    }
53
54    pub fn export_canvas(
55        &self,
56        canvas: &Canvas,
57        format: ExportFormat,
58        path: &str,
59    ) -> FlowResult<()> {
60        match format {
61            ExportFormat::Svg => {
62                let svg = self.export_canvas_to_svg(canvas)?;
63                let mut file = File::create(path)?;
64                file.write_all(svg.as_bytes())?;
65            }
66            ExportFormat::Json => {
67                let json = self.export_canvas_to_json(canvas)?;
68                let mut file = File::create(path)?;
69                file.write_all(json.as_bytes())?;
70            }
71            _ => {
72                return Err(FlowError::ExportError("Format not implemented".to_string()));
73            }
74        }
75        Ok(())
76    }
77
78    pub fn export_canvas_to_svg(&self, canvas: &Canvas) -> FlowResult<String> {
79        let renderer = DiagramRenderer::new();
80        renderer.render_to_svg(canvas)
81    }
82
83    pub fn export_canvas_to_json(&self, canvas: &Canvas) -> FlowResult<String> {
84        let json = serde_json::to_string_pretty(canvas)
85            .map_err(|e| FlowError::ExportError(format!("JSON serialization error: {}", e)))?;
86        Ok(json)
87    }
88}
89
90pub fn export_to_svg(canvas: &Canvas, path: &str) -> FlowResult<()> {
91    let exporter = FlowExporter::new();
92    exporter.export_canvas(canvas, ExportFormat::Svg, path)
93}
94
95pub fn export_to_json(canvas: &Canvas, path: &str) -> FlowResult<()> {
96    let exporter = FlowExporter::new();
97    exporter.export_canvas(canvas, ExportFormat::Json, path)
98}
