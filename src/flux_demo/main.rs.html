1//! Flux Vector Engine demo application
2
3use flux_vector_engine::{
4    capabilities,
5    graphics::{Color, FillType, Paint, Path, VectorEngine},
6    init,
7    ui::{
8        Animation, Button, Component, ComponentContext, EasingFunction, Event, EventType, ListView,
9        TextField,
10    },
11    VSyncMode, WindowConfig,
12};
13use std::time::Duration;
14
15fn main() -> Result<(), Box<dyn std::error::Error>> {
16    println!("Flux Vector Engine Demo");
17    println!("========================\n");
18
19    // Initialize the engine
20    init()?;
21
22    // Display engine capabilities
23    let caps = capabilities();
24    println!("Engine Capabilities:");
25    println!("  Version: {}", caps.version);
26    println!("  Max Texture Size: {}", caps.max_texture_size);
27    println!("  Max Framebuffer Size: {}", caps.max_framebuffer_size);
28    println!("  Supports Ray Tracing: {}", caps.supports_ray_tracing);
29    println!(
30        "  Supports Compute Shaders: {}",
31        caps.supports_compute_shaders
32    );
33    println!("  Max MSAA Samples: {}", caps.max_msaa_samples);
34    println!("  Max Anisotropy: {}", caps.max_anisotropy);
35    println!();
36
37    // Create window configuration
38    let config = WindowConfig {
39        width: 1920,
40        height: 1080,
41        title: "Flux Vector Engine Demo".to_string(),
42        refresh_rate: 120,
43        vsync: VSyncMode::Adaptive,
44        resizable: true,
45        decorations: true,
46        transparent: false,
47        always_on_top: false,
48    };
49
50    println!("Window Configuration:");
51    println!("  Resolution: {}x{}", config.width, config.height);
52    println!("  Title: {}", config.title);
53    println!("  Refresh Rate: {} Hz", config.refresh_rate);
54    println!("  VSync: {:?}", config.vsync);
55    println!();
56
57    // Create vector engine
58    let mut vector_engine = VectorEngine::new();
59
60    // Create a path
61    let mut path = Path::new();
62    path.move_to(100.0, 100.0);
63    path.line_to(200.0, 100.0);
64    path.quad_to(250.0, 150.0, 200.0, 200.0);
65    path.line_to(100.0, 200.0);
66    path.close();
67
68    vector_engine.add_path(path);
69
70    // Create a paint
71    let paint = Paint {
72        color: Color::rgb(0x2196F3),
73        fill_type: FillType::EvenOdd,
74        stroke_width: 2.0,
75        stroke_color: Color::rgb(0x1976D2),
76        ..Default::default()
77    };
78
79    vector_engine.add_paint(paint);
80
81    println!("Vector Engine:");
82    println!("  Paths: {}", vector_engine.paths().len());
83    println!("  Paints: {}", vector_engine.paints().len());
84    println!();
85
86    // Create UI widgets
87    let mut button = Button::new("Click Me".to_string()).with_bounds(100.0, 300.0, 120.0, 40.0);
88
89    let mut text_field =
90        TextField::new("Enter text...".to_string()).with_bounds(100.0, 360.0, 200.0, 30.0);
91
92    let mut list_view = ListView::new(vec![
93        "Item 1".to_string(),
94        "Item 2".to_string(),
95        "Item 3".to_string(),
96        "Item 4".to_string(),
97        "Item 5".to_string(),
98    ])
99    .with_bounds(100.0, 410.0, 200.0, 150.0);
100
101    println!("UI Widgets:");
102    println!("  Button: {}", button.text);
103    println!("  TextField: {}", text_field.placeholder);
104    println!("  ListView: {} items", list_view.items.len());
105    println!();
106
107    // Test event handling
108    let click_event = Event::new(EventType::Click, 150.0, 320.0);
109    button.handle_event(&click_event);
110    println!("Button clicked: {}", button.is_clicked());
111
112    let mouse_move_event = Event::new(EventType::MouseMove, 150.0, 320.0);
113    button.handle_event(&mouse_move_event);
114    println!("Button hovered: {}", button.hovered);
115
116    let list_click_event = Event::new(EventType::Click, 150.0, 430.0);
117    list_view.handle_event(&list_click_event);
118    println!("Selected item: {:?}", list_view.selected_item());
119    println!();
120
121    // Test animations
122    let animation = Animation::new(Duration::from_secs(1))
123        .with_easing(EasingFunction::EaseInOutCubic)
124        .with_keyframes(vec![(0.0, 0.0), (1.0, 100.0)]);
125
126    println!("Animation:");
127    println!("  Duration: {:?}", animation.duration);
128    println!("  Easing: {:?}", animation.easing);
129    println!("  Keyframes: {:?}", animation.keyframes);
130    println!();
131
132    // Test animation evaluation
133    for t in [0.0, 0.25, 0.5, 0.75, 1.0] {
134        let value = animation.evaluate(t);
135        println!("  t={:.2}: value={:.2}", t, value);
136    }
137    println!();
138
139    // Test all easing functions
140    println!("Easing Functions (t=0.5):");
141    let easing_functions = vec![
142        ("Linear", EasingFunction::Linear),
143        ("EaseIn", EasingFunction::EaseIn),
144        ("EaseOut", EasingFunction::EaseOut),
145        ("EaseInOut", EasingFunction::EaseInOut),
146        ("EaseInCubic", EasingFunction::EaseInCubic),
147        ("EaseOutCubic", EasingFunction::EaseOutCubic),
148        ("EaseInOutCubic", EasingFunction::EaseInOutCubic),
149        ("EaseOutBounce", EasingFunction::EaseOutBounce),
150        ("EaseOutElastic", EasingFunction::EaseOutElastic),
151    ];
152
153    for (name, easing) in easing_functions {
154        let value = easing.apply(0.5);
155        println!("  {}: {:.4}", name, value);
156    }
157    println!();
158
159    // Test color creation
160    let blue = Color::rgb(0x2196F3);
161    let red = Color::rgb(0xF44336);
162    let green = Color::rgb(0x4CAF50);
163    let transparent = Color::rgba(0x2196F3FF);
164
165    println!("Colors:");
166    println!(
167        "  Blue: R={}, G={}, B={}, A={}",
168        blue.r, blue.g, blue.b, blue.a
169    );
170    println!("  Red: R={}, G={}, B={}, A={}", red.r, red.g, red.b, red.a);
171    println!(
172        "  Green: R={}, G={}, B={}, A={}",
173        green.r, green.g, green.b, green.a
174    );
175    println!(
176        "  Transparent: R={}, G={}, B={}, A={}",
177        transparent.r, transparent.g, transparent.b, transparent.a
178    );
179    println!();
180
181    println!("Demo completed successfully!");
182
183    Ok(())
184}
